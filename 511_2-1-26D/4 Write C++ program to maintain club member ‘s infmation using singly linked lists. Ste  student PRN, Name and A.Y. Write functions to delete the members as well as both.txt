// Write a C++ program to maintain club membersâ€™ information using singly linked lists.
// Store student PRN, Name and A.Y.
// Write functions to delete the members as well as both coordinators.

#include <iostream>
#include <string>
using namespace std;

struct Node {
    int prn;
    string name;
    string ay;
    string role; // Coordinator / Member
    Node* next;
};

Node* head = NULL;

// Add coordinator at beginning
void addCoordinator(int prn, string name, string ay) {
    Node* newNode = new Node();
    newNode->prn = prn;
    newNode->name = name;
    newNode->ay = ay;
    newNode->role = "Coordinator";
    newNode->next = head;
    head = newNode;
}

// Add member at end
void addMember(int prn, string name, string ay) {
    Node* newNode = new Node();
    newNode->prn = prn;
    newNode->name = name;
    newNode->ay = ay;
    newNode->role = "Member";
    newNode->next = NULL;

    if (head == NULL) {
        head = newNode;
        return;
    }

    Node* temp = head;
    while (temp->next != NULL)
        temp = temp->next;

    temp->next = newNode;
}

// Delete a member by PRN
void deleteMember(int prn) {
    if (head == NULL)
        return;

    Node* temp = head;
    Node* prev = NULL;

    while (temp != NULL && temp->prn != prn) {
        prev = temp;
        temp = temp->next;
    }

    if (temp == NULL)
        return;

    if (prev == NULL)
        head = temp->next;
    else
        prev->next = temp->next;

    delete temp;
}

// Delete both coordinators (from beginning)
void deleteCoordinators() {
    while (head != NULL && head->role == "Coordinator") {
        Node* temp = head;
        head = head->next;
        delete temp;
    }
}

// Display list
void display() {
    Node* temp = head;
    while (temp != NULL) {
        cout << temp->prn << " "
             << temp->name << " "
             << temp->ay << " "
             << temp->role << endl;
        temp = temp->next;
    }
}

int main() {
    int prn, choice;
    string name, ay;

    do {
        cin >> choice;
        if (choice == 1) {
            cin >> prn >> name >> ay;
            addCoordinator(prn, name, ay);
        } else if (choice == 2) {
            cin >> prn >> name >> ay;
            addMember(prn, name, ay);
        } else if (choice == 3) {
            cin >> prn;
            deleteMember(prn);
        } else if (choice == 4) {
            deleteCoordinators();
        }
    } while (choice != 0);

    display();
    return 0;
}

/*
Explanation:
- Singly linked list stores club member information (PRN, Name, A.Y., Role).
- Coordinators are added at the beginning, members at the end.
- deleteMember() removes a member using PRN.
- deleteCoordinators() removes both coordinators from the start of the list.
- Display function prints the final list.
*/
