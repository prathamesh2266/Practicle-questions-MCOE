// Question 66:
// Write a C++ program to implement Linear Search.
// Given an array of integers, search a specific value entered by the user
// and print the index of the element if found, otherwise print "Element not found".

#include <iostream>
using namespace std;

int main() {
    int n;
    cin >> n;

    int arr[100];
    for (int i = 0; i < n; i++) {
        cin >> arr[i];
    }

    int key;
    cin >> key;

    int index = -1;

    // Linear Search
    for (int i = 0; i < n; i++) {
        if (arr[i] == key) {
            index = i;
            break;
        }
    }

    if (index == -1)
        cout << "Element not found";
    else
        cout << "Element found at index " << index;

    return 0;
}

/*
Explanation:
- Linear search checks each element one by one.
- If the element matches the key, its index is returned.
- If traversal completes without a match, the element is not found.
*/
