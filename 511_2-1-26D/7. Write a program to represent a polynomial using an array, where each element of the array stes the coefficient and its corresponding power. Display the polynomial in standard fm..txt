// Question 7:
// Write a program to represent a polynomial using an array, where each element of the
// array stores the coefficient and its corresponding power.
// Display the polynomial in standard form.

#include <iostream>
using namespace std;

struct Term {
    int coeff;
    int power;
};

int main() {
    int n;
    cin >> n;

    Term poly[50];

    for (int i = 0; i < n; i++) {
        cin >> poly[i].coeff >> poly[i].power;
    }

    cout << "Polynomial: ";

    for (int i = 0; i < n; i++) {
        if (poly[i].coeff == 0)
            continue;

        cout << poly[i].coeff << "x^" << poly[i].power;

        if (i != n - 1)
            cout << " + ";
    }

    return 0;
}

/*
Explanation:
- Polynomial is represented using an array of structures.
- Each structure stores coefficient and corresponding power.
- Input is taken as coefficientâ€“power pairs.
- Polynomial is displayed in standard mathematical form.
