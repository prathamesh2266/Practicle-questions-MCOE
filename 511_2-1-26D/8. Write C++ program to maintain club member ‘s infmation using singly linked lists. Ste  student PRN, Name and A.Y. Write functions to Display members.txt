// Question 8:
// Write a C++ program to maintain club members’ information using singly linked lists.
// Store student PRN, Name and A.Y.
// Write functions to display members.

#include <iostream>
#include <string>
using namespace std;

struct Node {
    int prn;
    string name;
    string ay;
    Node* next;
};

Node* head = NULL;

// Add member at end
void addMember(int prn, string name, string ay) {
    Node* newNode = new Node();
    newNode->prn = prn;
    newNode->name = name;
    newNode->ay = ay;
    newNode->next = NULL;

    if (head == NULL) {
        head = newNode;
        return;
    }

    Node* temp = head;
    while (temp->next != NULL)
        temp = temp->next;

    temp->next = newNode;
}

// Display members
void displayMembers() {
    Node* temp = head;
    while (temp != NULL) {
        cout << temp->prn << " "
             << temp->name << " "
             << temp->ay << endl;
        temp = temp->next;
    }
}

int main() {
    int prn, choice;
    string name, ay;

    do {
        cin >> choice;
        if (choice == 1) {
            cin >> prn >> name >> ay;
            addMember(prn, name, ay);
        }
    } while (choice != 0);

    displayMembers();
    return 0;
}

/*
Explanation:
- Singly linked list is used to store club members’ information.
- Each node stores PRN, Name, Academic Year, and next pointer.
- addMember() adds a member at the end of the list.
- displayMembers() traverses and prints all members.
*/
